SELECT
    OKC_K_HEADERS_ALL_B.CONTRACT_NUMBER,
    OKC_K_HEADERS_ALL_B.OBJECT_VERSION_NUMBER,
    PO_HEADERS_ALL.DOCUMENT_STATUS,
    OKC_K_HEADERS_ALL_B.CURRENCY_CODE,
    HZ_PARTIES.PARTY_UNIQUE_NAME,
    PO_HEADERS_ALL.COMMENTS,
    PER_PERSON_NAMES_F_V.LAST_NAME, PER_PERSON_NAMES_F_V.FIRST_NAME, PER_PERSON_NAMES_F_V.MIDDLE_NAMES
FROM
    PO_LINE_TYPES_TL, PO_LINES_ALL,
    PO_HEADERS_ALL, PER_PERSON_NAMES_F_V,
    PO_LINE_LOCATIONS_ALL,
    HZ_PARTIES, POZ_SUPPLIERS,
    OKC_K_HEADERS_ALL_B
WHERE 1=1
    AND (PO_LINE_TYPES_TL.LINE_TYPE_ID = PO_LINES_ALL.LINE_TYPE_ID)
    AND (PO_HEADERS_ALL.PO_HEADER_ID = PO_LINES_ALL.PO_HEADER_ID)
    AND (PO_HEADERS_ALL.PO_HEADER_ID = PO_LINE_LOCATIONS_ALL.PO_HEADER_ID)
    AND (PER_PERSON_NAMES_F_V.PERSON_ID=PO_HEADERS_ALL.AGENT_ID)
    AND (HZ_PARTIES.PARTY_ID = POZ_SUPPLIERS.PARTY_ID)
    AND (POZ_SUPPLIERS.VENDOR_ID = PO_HEADERS_ALL.VENDOR_ID)
    AND (PO_HEADERS_ALL.ATTRIBUTE5 = OKC_K_HEADERS_ALL_B.CONTRACT_NUMBER)
    AND (OKC_K_HEADERS_ALL_B.CURRENCY_CODE = 'RUR')
